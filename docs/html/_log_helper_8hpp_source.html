142template<typename List, typename ...Args>
143void log_list
144(
145 const std::string_view fileName,
146 const std::string_view funcName,
147 const size_t lineNo,
148 const List& msgList,
149 const std::string_view format_str,
150 Args&&... args
151)
152{
153 // If and only if, it is either a vector or std::list of strings
154 if constexpr (is_list<List>::value || is_vector<List>::value)
155 {
156 setLoggerProperties(fileName,
157 funcName,
158 FORWARD_ANGLES,
159 lineNo,
160 std::this_thread::get_id(),
161 LOG_TYPE::LOG_INFO);
162
163 logMsg(format_str, args...);
164 loggingOps << msgList;
165 }
166}

429template<typename ...Args>
430void log_assert
431(
432 const std::string_view fileName,
433 const std::string_view funcName,
434 const size_t lineNo,
435 const std::string_view cond,
436 const bool exitGracefuly,
437 const std::string_view format_str,
438 Args&&... args
439)
440{
441 if (cond.empty())
442 return;
443
444 loggerObj.setAssertCondition(cond); // Set the assertion condition
445 setLoggerProperties(fileName,
446 funcName,
447 FORWARD_ANGLE,
448 lineNo,
449 std::this_thread::get_id(),
450 LOG_TYPE::LOG_ASSERT);
451
452 logMsg(format_str, args...);
453 if (exitGracefuly)
454 {
455 loggingOps.~LoggingOps();
456 loggerObj.~Logger();
457 std::exit(EXIT_FAILURE);
458 }
459 else
460 {
461 std::abort();
462 }
463}
